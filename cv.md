def is_correct_bracket(text):
    brackets = []
    for char in text:
        if char == '(':
            brackets.append(char)
        elif char == ')':
            if not brackets:
                return False
            brackets.pop()
    return len(brackets) == 0

txt = input()

print(is_correct_bracket(txt))
